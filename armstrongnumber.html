Code to print armstrong number


        

        #include<stdio.h>

        using namespace std;

        int main()

        {  int num, r, sum=0, temp_num;
            printf("Enter number to check Armstrong number ");
            scanf("%d", &num);
            temp_num = num;
            while(num > 0)
            {
            r = num % 10;
            sum = sum + (r * r * r);
            num = num / 10;
            }
            if(temp_num == sum)
            {
            printf("Entered number is Armstrong Number.");
            printf("\n");
            }
            else
            {
            printf("Entered number is not Armstrong Number.");
            printf("\n");
            }
            return 0;
        }
